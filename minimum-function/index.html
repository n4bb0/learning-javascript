Minimum - function

Exercise from Eloquent JavaScript

Write a function that takes two arguments and returns their minimum.

--- Exercise below ---




My code:


      // html
      <div class="form">
        <label for="firstNum" class="label">Enter the first number</label>
        <input type="text" id="firstNum" class="numField">
        <br><br>
        <label for="secondNum" class="label">Enter the second number</label>
        <input type="text" id="secondNum" class="numField">
        <br>
        <input type="submit" value="submit" class="submitBtn">
      </div>
      <p id="result"></p>


      // javascript
      <script>
        var field1 = document.getElementById('firstNum');
        var field2 = document.getElementById('secondNum');
        var btn = document.querySelector('.submitBtn');
        var result = document.getElementById('result');

        function minimum (a, b) {
          if (a < b) {
            return a;
          } else {
            return b;
          }
        }

        btn.onclick = function() {
          var num1 = Number(field1.value);
          var num2 = Number(field2.value);
          if (isNaN(num1) || isNaN(num2)){
            result.style.backgroundColor = '';
            result.textContent = 'Please submit two numbers';
          } else {
            result.style.backgroundColor = 'rgba(255,0,0,0.5)';
            result.textContent = minimum(num1, num2) + ' is the smallest number';
          }
        }
      </script>

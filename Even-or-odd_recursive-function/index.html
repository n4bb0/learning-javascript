Even or odd

Using a recursive function

Exercise from Eloquent JavaScript

Define a recursive function isEven that accept a number parameter and return a Boolean.
The logic should be: zero is even, one is odd, for any other number N, its evenness is the same as N - 2.
Make it work with negative numbers too

My code:

  
      // html
      <div class="form">
        <label for="inputNmb">Enter an integer number between -10000 and 10000</label>
        <input type="text" id="inputNmb">
        <input type="submit" value="submit" id="submitNmb">
      </div>
      <p id="result"></p>

      // javascript
      <script>
        // number is even. true / false
       function isEven(number) {
         var absNumber = Math.abs(number)
         if (absNumber === 0){
           return true;
         } else if (absNumber === 1) {
           return false;
         } else {
           return isEven(absNumber - 2);
         }
       }

       var nmb = document.getElementById('inputNmb');
       var btn = document.getElementById('submitNmb');
       var result = document.getElementById('result');

       btn.onclick = function() {
         var userNmb = Number(nmb.value);
         if (isNaN(userNmb) || !Number.isInteger(userNmb) || userNmb > 10000 || userNmb < -10000) {  // check if the number is valid
           result.textContent = 'Please enter a valid number';
         } else {
           if (isEven(userNmb)) {
             result.textContent = userNmb + ' is even';
           } else {
             result.textContent = userNmb + ' is odd';
           }
         }
       }
      </script>
